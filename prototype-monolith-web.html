/* Hero occupa ~85% del viewport mobile */ @media (max-width: 768px) { .slot-wide:first-child { min-height: 85svh; display: flex; flex-direction: column; justify-content: flex-end; padding-bottom: 2rem; } }
Mobile — Scroll Snap Skills (globals.css)
/* Skills: scroll orizzontale con snap */ @media (max-width: 768px) { .skills-list { display: flex; overflow-x: auto; scroll-snap-type: x mandatory; -webkit-mask-image: linear-gradient( to right, black 75%, transparent 100%); gap: 8px; } .skills-list > * { flex: 0 0 auto; scroll-snap-align: start; } }
